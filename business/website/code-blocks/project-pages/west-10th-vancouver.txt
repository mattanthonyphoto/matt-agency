<h2>Contemporary, sleek design that maximizes natural light and expansive urban views</h2>
<p>The W 10th project in Vancouver features a contemporary, sleek design that maximizes natural light and provides expansive views of the surrounding urban landscape. The residence utilizes Reynaers' Masterline 8 for the windows, MasterPatio for the sliding doors, and CW 50 for a custom skylight, offering not only beauty but also exceptional thermal efficiency.</p>

<script>
(function(){
  var isMobile = 'ontouchstart' in window;

  /* ═══ LOADING SCREEN ═══ */
  var loader = document.getElementById('tpLoader');
  window.addEventListener('load', function(){
    setTimeout(function(){ loader.classList.add('tp-done'); }, 1100);
  });

  /* ═══ SCROLL PROGRESS ═══ */
  var progressBar = document.getElementById('tpProgress');
  function updateProgress(){
    var s = window.scrollY;
    var d = document.documentElement.scrollHeight - window.innerHeight;
    if(d > 0) progressBar.style.width = (s / d * 100) + '%';
  }

  /* ═══ HERO PARALLAX ═══ */
  var hero = document.querySelector('.tp-hero');
  var heroBg = hero ? hero.querySelector('img') : null;
  var heroContent = hero ? hero.querySelector('.tp-hc') : null;
  function updateHero(){
    var s = window.scrollY;
    var h = window.innerHeight;
    if(s > h) return;
    var ratio = s / h;
    if(heroBg) heroBg.style.transform = 'translateY(' + (s * 0.35) + 'px) scale(' + (1 + ratio * 0.05) + ')';
    if(heroContent){
      heroContent.style.transform = 'translateY(' + (s * 0.15) + 'px)';
      heroContent.style.opacity = 1 - ratio * 1.2;
    }
  }

  /* ═══ CUSTOM CURSOR ═══ */
  var cursor = document.getElementById('tpCursor');
  var cursorDot = document.getElementById('tpCursorDot');
  var cx = 0, cy = 0, dx = 0, dy = 0;
  if(!isMobile && cursor){
    document.addEventListener('mousemove', function(e){
      dx = e.clientX; dy = e.clientY;
      cursorDot.style.left = dx + 'px';
      cursorDot.style.top = dy + 'px';
      if(!cursor.classList.contains('tp-active')){
        cursor.classList.add('tp-active');
        cursorDot.classList.add('tp-active');
      }
    });
    function tickCursor(){
      cx += (dx - cx) * 0.12;
      cy += (dy - cy) * 0.12;
      cursor.style.left = cx + 'px';
      cursor.style.top = cy + 'px';
      requestAnimationFrame(tickCursor);
    }
    tickCursor();
    var hoverEls = document.querySelectorAll('a, button, .tp-cb, .tp-nx, .tp-gg div');
    hoverEls.forEach(function(el){
      el.addEventListener('mouseenter', function(){ cursor.classList.add('tp-hover'); });
      el.addEventListener('mouseleave', function(){ cursor.classList.remove('tp-hover'); });
    });
    document.addEventListener('mousedown', function(){ cursor.classList.add('tp-click'); });
    document.addEventListener('mouseup', function(){ cursor.classList.remove('tp-click'); });
  }

  /* ═══ COMBINED SCROLL HANDLER ═══ */
  var ticking = false;
  window.addEventListener('scroll', function(){
    if(!ticking){
      requestAnimationFrame(function(){
        updateProgress();
        updateHero();
        ticking = false;
      });
      ticking = true;
    }
  }, {passive:true});

  updateProgress();
  updateHero();
})();
</script>
